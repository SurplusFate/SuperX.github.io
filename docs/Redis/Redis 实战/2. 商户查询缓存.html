    public <R, ID> R queryWithMutex(String keyPrefix, ID id, Function<ID, R> dbFallback,

        // 1.åˆ¤æ–­ redis æ˜¯å¦æœ‰ç¼“å­˜
        String key = keyPrefix + id;

        if (StrUtil.isNotBlank(json)) {
